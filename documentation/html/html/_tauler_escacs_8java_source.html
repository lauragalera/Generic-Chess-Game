772  private Map<Posicio, Moviment> trobarDestiCombinat(Posicio origen, Moviment mov) {
773  int incrF = 1; //increment que es suma la fila
774  int incrC = 1; //increment que es suma a la columna
775  int maxIncrF = _fila - 1; //increment màxim de fila, considerant el pitjor cas.
776  int maxIncrC = _columna - 1; //increment màxim de columna, considerant el pitjor cas.
777  if (!mov.fila().equals("n") && !mov.fila().equals("m")) { //si és -m, -n o un número...
778  incrF = -1; //l'increment és negatiu
779  if (!mov.fila().equals("-n") && !mov.fila().equals("-m")) { //si es tracta d'un número...
780  incrF = Integer.parseInt(mov.fila()); //l'increment és el número
781  maxIncrF = Math.abs(incrF); //el màxim de files que s'avancen és l'increment en valor absolut
782  }
783  }
784  //Fem el mateix per a les columnes...
785  if (!mov.columna().equals("n") && !mov.columna().equals("m")) {
786  incrC = -1;
787  if (!mov.columna().equals("-n") && !mov.columna().equals("-m")) {
788  incrC = Integer.parseInt(mov.columna());
789  maxIncrC = Math.abs(incrC);
790  }
791  }
792  int antIncF = incrF; //Guardem a una variable l'increment inicial de la fila
793  int antIncCol = incrC; //Guardem a una variable l'increment inicial de la columna
794  boolean fiColm = false; //Variable per saber si no queden més columnes
795  boolean fiFiles = false; //Variable per saber si no queden més files
796  Map<Posicio, Moviment> destins = new HashMap<>();
797  if (!(origen.fila() + incrF > _fila || origen.fila() + incrF < 1 || origen.columna() + incrC > _columna || origen.columna() + incrC < 1)) { //si sumant l'increment a la fila o la columna la posició ja no existeix vol dir que no hi ha cap destí
798  while (Math.abs(incrC) <= maxIncrC && !fiColm) { //Mentre no haguem arribat al maxim de columnes avançades i encara quedin columnes...
799  while (Math.abs(incrF) <= maxIncrF && !fiFiles) { //Mentre no haguem arribat al maxim de files avançades i encara quedin files...
800  Posicio posAct = new Posicio(origen.columna() + incrC, origen.fila() + incrF); //nova posició
801  if (!(existeixPos(posAct))) //si no existeix la posició...
802  fiFiles = true; //no podem seguir
803  else if (esPossibleAnarDesti(origen, posAct, mov)) { //La nova posició existeix però podem anar-hi segons mov.movCaptura() i el tauler?
804  destins.put(posAct, mov); //s'afegeix el destí amb el moviment.
805  }
806  incrF = incrF + antIncF; //avancem a la següent fila
807  if(!filaColDinsLimit(origen.columna()+incrC, origen.fila()+incrF)) fiFiles = true; //la següent fila no es troba dins els limits de posició
808  }
809  incrF = antIncF; //tornem a començar pel mateix increment de la fila
810  fiFiles = false;
811  incrC = incrC + antIncCol; //avancem a la següent columna
812  if (origen.columna() + incrC > _columna || origen.columna() + incrC < 1) //si voliem accedir a una columna que no existeix...
813  fiColm = true; //no podem seguir
814  }
815  }
816  return destins;
817  }
